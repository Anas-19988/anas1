import { financialModel } from "./models/financialModel.js"; import { classifyAccount } from "./financial-engine/classify.js"; import { computeIncome } from "./financial-engine/computeIncome.js"; import { computeBalance } from "./financial-engine/computeBalance.js"; import { computeRatios } from "./financial-engine/computeRatios.js"; import { validateModel } from "./financial-engine/validate.js"; export async function generateFinancialStatements() { if (trialBalanceProcessed.length === 0) { showUIMessage("لا توجد بيانات", "يرجى رفع ميزان المراجعة أو تجربة أحد الديموهات", "danger"); return; } try { showUIMessage("جاري المعالجة...", "يتم حساب القوائم المالية وفق معايير IFRS مع دعم الزكاة...", "success"); // إنشاء نسخة جديدة من النموذج const model = structuredClone(financialModel); // تصنيف الحسابات trialBalanceProcessed.forEach((acc, i) => { const select = document.getElementById(`map-acc-${i}`); const category = select ? select.value : "operating_expenses"; classifyAccount(acc, category, model); }); // حساب قائمة الدخل computeIncome(model); // حساب الميزانية computeBalance(model); // حساب النسب computeRatios(model); // التحقق من الأخطاء const errors = validateModel(model); if (errors.length > 0) { showUIMessage("خطأ", errors.join(" — "), "danger"); return; } // ربط النتائج مع دوال الواجهة الحالية fillIncomeStatement({ revenue: model.income.revenue, cogs: model.income.cogs, grossProfit: model.income.revenue - model.income.cogs, totalOpExpenses: model.income.totalExpenses, netIncomeBefore: model.income.netBeforeZakat, expenseDetails: model.income.expenses, zakat: model.income.netBeforeZakat * 0.025, tax: 0, netFinal: model.income.netAfterZakat }); fillBalanceSheet( model.totals.currentAssets, model.totals.fixedNet, model.totals.totalAssets, { cashAndEquivalents: model.assets.cash, receivables: model.assets.receivables, inventory: model.assets.inventory, currentLiabilities: model.liabilities.current, nonCurrentLiabilities: model.liabilities.nonCurrent }, model.totals.totalLiabilities, model.totals.totalEquity, model.totals.totalLiabilities + model.totals.totalEquity, model.income.netBeforeZakat ); updateKPICards( model.income.revenue, model.income.revenue - model.income.cogs, model.income.netAfterZakat, model.totals.totalAssets ); computeFinancialRatios( model.ratios.currentRatio, model.ratios.debtRatio, model.ratios.roa, model.ratios.netMargin ); renderExecutiveCharts({ revenue: model.income.revenue, cogs: model.income.cogs, grossProfit: model.income.revenue - model.income.cogs, totalExpenses: model.income.totalExpenses, netIncome: model.income.netAfterZakat, cashAndEquivalents: model.assets.cash, receivables: model.assets.receivables, inventory: model.assets.inventory, fixedAssets: model.totals.fixedNet, currentRatio: model.ratios.currentRatio, debtRatio: model.ratios.debtRatio, netMargin: model.ratios.netMargin, roa: model.ratios.roa }); checkBalance(model.totals.totalAssets, model.totals.totalLiabilities + model.totals.totalEquity); document.getElementById('statements-section').classList.remove('hidden'); document.getElementById('report-generation-date').textContent = new Date().toLocaleDateString('ar-SA', { year: 'numeric', month: 'long', day: 'numeric' }); updateStepIndicator(3); switchTab('analytics'); document.getElementById('statements-section').scrollIntoView({ behavior: 'smooth' }); showUIMessage("تم بنجاح", "تم توليد القوائم المالية بالكامل مع الزكاة والتحليل", "success"); } catch (error) { console.error("خطأ في توليد القوائم:", error); showUIMessage("خطأ", "حدث خطأ أثناء معالجة البيانات", "danger"); } }
